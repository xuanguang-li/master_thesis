\begin{frame}{Motivation}
	%content: motivation
	\begin{itemize}
		\item Investment is the most volatile component of the business cycle; a long literature treats \textbf{investment shocks} as a main driving force.
		\item \textbf{Justiniano, Primiceri \& Tambalotti (2010):} an estimated DSGE attributes a large share of U.S.\ fluctuations to a \emph{supply-side} investment-specific technology (IST) shock.
		\item \textbf{The comovement puzzle:} supply-side IST shocks struggle to reproduce the joint positive comovement of consumption, investment, and inflation seen in the data.
		\item \textbf{Nirei (2015):} an \emph{investment demand} (ID) shock---built from firm heterogeneity and lumpy investment---\emph{can} generate this comovement, but lacks empirical evidence.
	\end{itemize}
	\vspace{0.5em}
	\begin{block}{This paper}
		Estimate a DSGE model with \textbf{both} an IST shock and an investment demand shock to quantify how important the investment demand shock is for the U.S.\ business cycle.
	\end{block}
\end{frame}

\begin{frame}{Specifying the investment demand shock}
	%content: specification of the investment demand shock + two-step equilibrium
	\begin{itemize}\small
		\item The micro origin (idiosyncratic productivity $+$ lumpy investment) appears \emph{at the aggregate level} as a deviation of investment from its shock-free decision $\Rightarrow$ a reduced-form shock $\mu_{d,t}$:
		\[
			I_t=\mu_{d,t}\,I_t^{*},\qquad
			\ln\mu_{d,t}=(1-\rho_d)\ln\mu_d+\rho_d\ln\mu_{d,t-1}+\varepsilon_{d,t}.
		\]
		\item Households would otherwise undo the shock, so the equilibrium is solved in \textbf{two steps}:
		\begin{itemize}\footnotesize
			\item[\textbf{1.}] solve the baseline model (no ID shock) for the endogenous investment $I_t^{*}$;
			\item[\textbf{2.}] reveal $\mu_{d,t}$, \emph{fix} $I_t=\mu_{d,t}I_t^{*}$, and re-solve the equilibrium for consumption, labor, bonds and utilization.
		\end{itemize}
		\item Natural output for the Taylor rule is computed from \textbf{frictionless parallel models} in each stage (next slide).
	\end{itemize}
\end{frame}

\begin{frame}{A toy model of idiosyncratic shocks and lumpy investment}
	
	\textbf{Why is it legitimate to use a reduced-form investment demand shock?}
	\vspace{1em}

	Standard New-Keynesian CES aggregator and demand:
	\[
		Y_t=\Big(\int_0^1 y_{jt}^{\frac{\eta-1}{\eta}}\,dj\Big)^{\frac{\eta}{\eta-1}},
		\qquad
		y_{jt}=\Big(\frac{p_{jt}}{P_t}\Big)^{-\eta}Y_t,
		\qquad
		y_{jt}=a_{jt}k_{jt}.
	\]
	Linear-in-capital technology $y_{jt}=a_{jt}k_{jt}$. Combining demand and production, real revenue is
	\[
		r_{jt}(a_{jt},k_{jt})=\frac{p_{jt}y_{jt}}{P_t}
		=\big(a_{jt}k_{jt}\big)^{1-\frac1\eta}\,Y_t^{\frac1\eta}.
	\]
	\begin{itemize}\small
		\item Concave in own capital ($\eta>1$): diminishing returns through the demand curve.
		\item Revenue, hence the desired capital, scales with aggregate output $Y_t$ — the channel through which one firm's action will later spill over to all firms.
	\end{itemize}
\end{frame}

\begin{frame}{Lumpy investment and the investment threshold}
	% --- top region: text (left) + figure pinned NE (right) ---
	\begin{columns}[T,onlytextwidth]
		\column{0.58\textwidth}
		Firms solve
		\[\small
			\max_{\{x_{t+k}\}}\ \mathbb{E}_t\!\sum_{k\ge0}\Lambda_{t,t+k}\big(r_{j,t+k}-x_{j,t+k}\big),
		\]
		$$\begin{aligned}
			\text{s.t. }&k_{j,t+1}=(1-\delta)k_{j,t}+x_{j,t},\\
			&x_{j,t+k}\in\{0,\ \lambda k_{j,t+k}\}.
		\end{aligned}$$

		\column{0.4\textwidth}
		\vspace{-0.6em}
		\centering

			\draw[->] (0,0) -- (3.4,0) node[below]{$k_{t+1}$};

				plot ({\x},{3*exp(-2*(ln(\x)-0.459)^2)});

			\draw[dotted] (1,0) -- (1,1.97);
			\draw[dotted] (2.5,0) -- (2.5,1.97);
			\draw[dotted] (1.58,0) -- (1.58,3.0);

			\fill[invest] (1,1.97) circle (1.3pt);

			\node[below,font=\scriptsize] at (1,0) {$k^{*}$};

	\end{columns}

	% --- bottom region: full-width text wrapping under the figure ---
	\vspace{0.3em}
	Indifference at $k^*$ (the firm knows its next-period technology) pins down the threshold:
	\[
		k_{j,t+1}^{*}=a_{j,t+1}^{\,\eta-1}\,\Phi_t(\Lambda_t, \Lambda_{t+1})\,Y_{t+1}\,
	\]
	\begin{itemize}\small
		\item Desired capital scales with aggregate output $Y_{t+1}$ — the spillover channel that will carry one firm's action to all firms.
	\end{itemize}
	\vspace{0.4em}
	Normalize a firm's position by its \textbf{distance to the threshold}
	\[
		s_{jt}=\frac{\log\!\big(k_{jt}/k_{jt}^{*}\big)}{\log\lambda},
	\]
	so that investing means jumping up exactly one notch ($s\mapsto s+1$) in these units.
\end{frame}


\begin{frame}{Investing fraction and idiosyncratic shock}
	% --- top region: text (left) + figure pinned NE (right) ---
	\begin{itemize}
		\item Assume $a_t\in\{a_L,a_H\}$ and $s_t\mid a_t\sim U[0,1)$ under stationary distribution.
		\item Macro variables are constant: $Y = Y_t = Y_{t+1}$, $\Phi = \Phi_t = \Phi_{t+1}$
	\end{itemize}

	\begin{columns}[T,onlytextwidth]
		\column{0.56\textwidth}

		Without investing, under stationary distribution position drifts by
		\[
		\begin{aligned}
		s_{j,t+1}&=s_{j,t}(a) \\
		&-\underbrace{\frac{-\log(1-\delta)+(\eta-1)\log\frac{a'}{a}}{\log\lambda}}_{s^{*}(a,a')}.
		\end{aligned}
		\]

		\column{0.42\textwidth}
		\centering
		\begin{tikzpicture}[xscale=3.2,yscale=1.1]
			% axis
			\draw[->] (-0.05,0) -- (1.25,0) node[right,font=\scriptsize]{$s_t$};
			\draw[->] (0,-0.05) -- (0,1.9) node[above,font=\scriptsize]{density};
			% uniform density box on [0,1)
			\fill[accent!12] (0,0) rectangle (1,1.2);
			\draw[accent,thick] (0,1.2) -- (1,1.2);
			\draw[accent,thick] (0,0)--(0,1.2);
			\draw[accent,thick,dashed] (1,0)--(1,1.2);
			% investing region [0,s*]
			\fill[invest!30] (0,0) rectangle (0.35,1.2);
			\draw[invest,thick] (0.35,0)--(0.35,1.45);
			\node[invest,font=\scriptsize,above] at (0.35,1.45) {$s^{*}$};
			% epsilon shifted threshold
			\draw[shockc,thick,->] (0.35,0.55)--(0.55,0.55);
			\draw[shockc,thick,dashed] (0.55,0)--(0.55,1.2);
			\node[shockc,font=\scriptsize,above] at (0.62,1.18) {$s^{*}\!+\!\nu$};
			% labels
			\node[invest,font=\scriptsize] at (0.17,0.55) {invest};
			\node[font=\scriptsize] at (0.78,0.55) {stay};
			\node[below,font=\scriptsize] at (0,0) {$0$};
			\node[below,font=\scriptsize] at (1,0) {$1$};
		\end{tikzpicture}
	\end{columns}

	% --- bottom region: full-width text wrapping under the figure ---
	\vspace{0.3em}
	\begin{itemize}
		\item Since $s_t\sim U[0,1)$, $s^{*}(a,a')$ is the \textbf{investing fraction}
		\item Add an i.i.d.\ shock $\varepsilon$ on $a'$ \emph{only} in group $(a,a')=(b,b')$:
		\[\small
			s^{*}(b,\varepsilon b')=s^{*}(b,b')+\underbrace{\frac{(\eta-1)\log\varepsilon}{\log\lambda}}_{\nu}.
		\]
		\item The cutoff in that group rises by $\nu$, so an extra $\nu$-mass invests.
	\end{itemize}
\end{frame}

\begin{frame}{Propagation through the threshold}
	% --- top region: text (left) + figure pinned NE (right) ---
	\begin{columns}[T,onlytextwidth]
		\column{0.63\textwidth}
		More investment in $(b,b')$ raises \emph{future output} of this group, which is a part of \emph{future aggregate output}:
		\[\begin{aligned}
			Y_{t+1}(a, a', \nu)
			&=\Big(\omega(b,b')\!\!\int_{0}^{s^*+\nu}\!\! \big(b'\lambda (1-\delta) k_{t}(s_t)\big)^{\frac{\eta-1}{\eta}}\,dF(s_t\mid b) \\
			&+ \text{no investment firms}\Big)^{\frac{\eta}{\eta-1}}
		\end{aligned}
		\]

		\column{0.35\textwidth}
		\vspace{-0.3em}
		\centering
		\begin{tikzpicture}[scale=0.72]
			% 2x2 groups, sizes ~ omega
			\fill[gray!12] (0,0) rectangle (1.4,1.4);
			\fill[gray!12] (1.4,0) rectangle (2.6,1.4);
			\fill[gray!12] (0,1.4) rectangle (1.4,2.6);
			% highlight shocked group (b,b'): a=H (top), a'=H (right)
			\fill[shockc!25] (1.4,1.4) rectangle (2.6,2.6);
			\draw[shockc,very thick] (1.4,1.4) rectangle (2.6,2.6);
			% grid
			\draw (0,0) rectangle (2.6,2.6);
			\draw (1.4,0)--(1.4,2.6);
			\draw (0,1.4)--(2.6,1.4);
			% labels of omega
			\node[font=\scriptsize] at (0.7,0.7) {$\omega(L,L)$};
			\node[font=\scriptsize] at (2.0,0.7) {$\omega(L,H)$};
			\node[font=\scriptsize] at (0.7,2.0) {$\omega(H,L)$};
			\node[font=\scriptsize,shockc] at (2.0,2.0) {$\omega(b,b')$};
			% axis labels
			\node[below,font=\scriptsize] at (0.7,0) {$a'=L$};
			\node[below,font=\scriptsize] at (2.0,0) {$a'=H$};
			\node[left,font=\scriptsize] at (0,0.7) {$a=L$};
			\node[left,font=\scriptsize] at (0,2.0) {$a=H$};
		\end{tikzpicture}
	\end{columns}

	% --- bottom region: full-width text wrapping under the figure ---
	\vspace{0.5em}
	\begin{itemize}
		\item Higher $Y_{t+1}$ shifts the cutoff in \emph{every} group:
		\[
		\begin{aligned}
			\nu^{(1)}(a, a') 
			&\coloneq \Delta s^{*}_t(a,a',\nu)\\
			&=s^*(a, a') + \frac{\log (Y_{t+1}(\nu)/ Y)}{\log\lambda} - \left(s^*(a, a') + \frac{\log (\overset{=Y}{Y_{t+1}(0)}/ Y)}{\log\lambda}\right)
		\end{aligned}
		\]
		\item aggregate \textbf{$\nu$-induced fraction}:
		\[
			\nu^{(1)}\coloneq \sum_{(a,a')}\omega(a,a')\,\nu^{(1)}(a, a').
		\]
		\item The first $\nu$ of investing firms raises $Y$, which induces $\nu^{(1)}$ more, which raises $Y$ again $\dots$ a GE multiplier that iterates to a new equilibrium.
	\end{itemize}
\end{frame}

\begin{frame}{Total induced investment $\gg \nu$}
	$$\text{Total induced investment} = \sum_{i=1}^{\infty} \text{ induced fraction}^{(i)} \times \text{ average induced investment}^{(i)}$$

	\begin{itemize}
		\item Scale of induced fraction
		\[
		 \lim_{\nu\to 0}\frac{\nu^{(1)}}{\omega(b,b')\,\nu}=1\
		\]
		\begin{itemize}
			\small
			\item the $\nu^{(1)}$ has the \emph{same scale} as the original fraction 
			
			$\Rightarrow$ each $\omega(a, a') \nu^{(1)}(a, a')$ will induce the \emph{same scale} $\nu^{(2)}(\nu^{(1)}(a, a'))$ again 
		
			$\Rightarrow$ induced fraction will not decay through iteration.
		\end{itemize}

		\item Scale of average induced investment
		\[
			\frac{\partial \Delta x(a, a', \nu)}{\partial \nu}\Big|_{\nu=0} = (\lambda - 1)(1-\delta) \lambda^{s^*} a^{\eta - 1}\Psi Y
		\]
		\begin{itemize}
			\small
			\item $\nu$-induced average investment, $\Delta x(a, a', \nu)$, is defined as $\Delta x_t(a, a', \nu) \coloneq \int_{s^*}^{s^*+\nu} (\lambda - 1)(1-\delta) \underbrace{\lambda^{s_t} a^{\eta-1}\Phi Y}_{k_t(s_t)} \, d F(s_t\,|\, a_t = a)$
			
			$\Rightarrow$ the derivative says $\Delta x(a, a', \nu) = O(\nu)$
			
			$\Rightarrow$ average induced investment will not decay faster than $\nu$.
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{The estimated DSGE model}
	% list the features of the model used in estimation, like sticky wage, investment adjustment cost, etc.
	Based on Smets--Wouters (2007) and Herbst--Schorfheide (2015), augmented with the investment demand shock (two-step equilibrium).
	\begin{itemize}\small
		\item \textbf{Nominal rigidities:} Calvo price setting (no indexation); reduced-form \textbf{sticky wage}---partial adjustment of the wage toward the MRS:
		\[
			w_t=\Big(L_t^{\sigma_\ell}\big/C_t^{-\sigma}\Big)^{\theta_w}\,w^{\,1-\theta_w}.
		\]
		\item \textbf{Real frictions:} \textbf{investment adjustment cost} $S(I_t/I_{t-1})$ and \textbf{endogenous capital utilization} $u_t$. Capital law of motion carries the IST shock:
		\[
			K_{t+1}=(1-\delta)K_t+\mu_{s,t}\big[1-S(I_t/I_{t-1})\big]I_t.
		\]
		\item \textbf{Policy:} Taylor rule (inflation $+$ output gap $+$ smoothing); output-stabilizing government spending.
		\item \textbf{Five structural shocks:} technology $\varepsilon_a$, IST $\varepsilon_s$, monetary $\varepsilon_R$, government $\varepsilon_g$, and investment demand $\varepsilon_d$.
	\end{itemize}
\end{frame}

\begin{frame}{Estimation: method and data}
	% 1. brief introduction on estimation method
	% 2. Data
	\begin{columns}[T,onlytextwidth]
		\column{0.5\textwidth}
		\textbf{Method}
		\begin{itemize}\small
			\item Bayesian estimation; methodology follows Smets--Wouters (2007) and Justiniano et al.\ (2010).
			\item Posterior via Metropolis--Hastings: $100{,}000$ draws, $0.5$ burn-in, relatively flat priors.
			\item \textbf{ID vs.\ IST identification} rests on their different \emph{comovement} signatures---there is no direct data series for the ID shock.
		\end{itemize}
		\column{0.5\textwidth}
		\textbf{Data}
		\begin{itemize}\small
			\item Six quarterly U.S.\ observables (FRED), \textbf{1983:Q1--2002:Q4}:
			\begin{itemize}\footnotesize
				\item output growth, CPI inflation, fed funds rate,
				\item hours, investment growth, gov.-spending growth.
			\end{itemize}
			\item Sample fits a constant linear trend, is comparable to JPT, and avoids the ZLB.
			\item Measurement error ($4\%$ of each series' variance) avoids stochastic singularity.
		\end{itemize}
	\end{columns}
\end{frame}

\begin{frame}{Baseline result: variance decomposition}

		\column{0.5\textwidth}
		\footnotesize\textcolor{accent}{\textbf{IST shock $\Rightarrow$ deflationary}}
		\begin{itemize}\scriptsize
			\item Wage $\downarrow$ while hours $\uparrow$, consumption $\downarrow$, output $\approx$ flat.
			\item Strong \emph{intertemporal substitution}: cheap investment $\Rightarrow$ save more today (labor supply shifts out).
		\end{itemize}
		\column{0.5\textwidth}
		\footnotesize\textcolor{invest}{\textbf{ID shock $\Rightarrow$ inflationary}}
		\begin{itemize}\scriptsize
			\item Wage and hours rise \emph{together} $\Rightarrow$ labor \emph{demand} shift dominates; output rises strongly.
			\item \emph{Wealth effect}: higher future output raises lifetime wealth, so little substitution---hence comovement.
		\end{itemize}
	\end{columns}

\begin{frame}{Robustness: Smets--Wouters-style model}
	% FEVD and IRF of SW-type model
	Add habit formation, a monopolistic labor market (Erceg wage Phillips curve), and preference / price- / wage-markup shocks; add consumption and wage growth as observables.
	\begin{itemize}\footnotesize
		\item \textbf{Better hours fit:} $\mathrm{Var}(\hat N_t)$ falls $32.2\!\to\!9.8$ (data $6.3$)---the \textbf{wage-markup shock} does the work.
		\item \textbf{ID shock turns marginal:} only $4.4\%$ of output and $13.6\%$ of investment FEVD.
		\item \textbf{IRFs flip:} IST is now \emph{inflationary} with a \emph{positive} wage response; the ID consumption response is far smoother ($\hat c_1$ is $0.2\%$ of $\hat\imath_1$, vs.\ $12\%$ in the baseline).
	\end{itemize}

\begin{frame}{Conclusion}
	% Conclusion
	\begin{itemize}
		\item Revisits Justiniano et al.\ (2010) by testing Nirei's \textbf{investment demand shock} in an estimated DSGE with \emph{both} IST and ID shocks.
		\item \textbf{Baseline:} the ID shock alone explains $\approx 40\%$ of output and $\approx 50\%$ of investment FEVD---a competitive driver---and generates an \emph{inflationary} expansion with comovement (vs.\ deflationary IST).
		\item \textbf{Limitations / fragility:}
		\begin{itemize}\small
			\item no direct data identifies the ID shock---identification leans on comovement differences, hard to test over the whole parameter space;
			\item imperfect fit and some extreme posteriors in the baseline;
			\item results are specification-sensitive: the SW-style model shrinks the ID shock, and an i.i.d.\ ID shock weakens it.
		\end{itemize}
		\item \textbf{Takeaway:} structural estimation gives genuine insight into investment-side shocks, but the model specification must be chosen carefully.
	\end{itemize}
\end{frame}
